Triggers attached custom events

 
		trigger : function (type, args) {
			if (!this.vars.events) {
				return;
			}
			
			var events = this.vars.events["on" + type],
			    i, j, event;
			
			if (events && events.length) {
				for (i = 0, j = events.length; i < j; i++) {
					event = events[i];
					
					if (event) {
						event.apply(this, args || [{
							type : type
						}]);
					}
				}
			}
		}
		
	});
